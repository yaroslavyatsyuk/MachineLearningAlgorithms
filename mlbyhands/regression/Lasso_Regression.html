LASSO REGRESSION IMPLEMENTATION¶

COORDINATE DESCENT ALGORITHM¶

Normalized features version¶

$$w = 0$$¶

$$for j = 0,1,...,D:$$¶

$$p_{j} = \sum \limits _{i=1}^N x_{j}^{(i)}(y^{(i)} - \hat{y}_{pred}^{(i)} + w_{j}x_{j}^{(i)})$$¶

$$ \hat{w}_j= \begin{cases} (p_j +\alpha) & \text{ if } p_j < -\alpha/2\\ 0 & \text{ if } p_j \ in \ [-\alpha/2, \alpha/2]\\ (p_j -\alpha) & \text{ if } p_j > \alpha/2 \end{cases} $$

class LassoRegression:
    """
    Coordinate descent for lasso regression - normalized data version
    
    n_iterations: float
        The number of iteration witch algorithm will perform for tuning weights,
        this attribute works only for gradient_descent solver
    learning_rate: float
        The gradient decent step size, this attribute works only for gradient_descent solver
    alpha: float
       Regularization coefficient, higher lamdba -> higher regularization, must be a positive float.
       default = 1e-3
    """

    def __init__(self, alpha=.01, n_iterations=100, features_names=None, verbose=False):
        self.alpha = alpha
        self.verbose = verbose
        self.n_iterations = n_iterations
        self.features_names = features_names

    def fit(self, X, y):
        X = np.array(X)
        y = np.array(y)

        X = np.insert(np.array(X), 0, 1, axis=1)
        X, self.norms = self._normalize_features(X)

        self.weights = np.zeros(X.shape[1])

        self.weights = self._coordinate_descent_lasso(
            X, y, self.weights, self.alpha, self.n_iterations, self.verbose)

    def _coordinate_descent_lasso(self, X, y, w, alpha, n_iterations, verbose):
        m, n = X.shape

        for i in range(n_iterations):

            # Looping through each coordinate
            for j in range(n):
                X_j = X[:, j]
                y_pred = X @ w

                # Equivalent to not using feature j in current iteration 
                # (y - y_pred + w[j] * X_j) - RSS without feature j
                rho = X_j @ (y - y_pred + w[j] * X_j)

                # Do not penalize intercept
                if (j == 0):
                    w[j] = rho
                else:
                    w[j] = self._soft_threshold(rho, alpha)
                    
        if (verbose):
            print("Lasso Regression Training Finished")
            print("Training MSE:", self._calculate_MSE(X, y, w))
            
        return w

    def _normalize_features(self, X):
        norms = np.linalg.norm(X, axis=0)
        normalized_features = X / norms
        return [normalized_features, norms]

    def _soft_threshold(self, rho, alpha):
        '''Soft threshold function used for normalized data and lasso regression'''
        if rho < - alpha:
            return (rho + alpha)
        elif rho > alpha:
            return (rho - alpha)
        else:
            return 0

    def _calculate_MSE(self, X, y, w):
        '''Mean Square Error'''
        m = y.shape[0]
        return np.sum((y - X @ w) ** 2) / m

    def get_weights(self):
        return self.weights

    def coefficients(self):
        if self.features_names == None:
            return self.weights
        else:
            features = ["intercept"] + self.features_names;
            return pd.DataFrame({"features": features, "coefficients": self.weights})

    def predict(self, X):
        X = np.insert(np.array(X), 0, 1, axis=1)
        return X @ (self.weights / self.norms)

theta_list = list()
alpha = np.logspace(3,8, 30) / 10 
for l1 in alpha:
    lasso = LassoRegression(alpha=l1, features_names=all_features)
    lasso.fit(train[all_features], train["price"])
    
    theta_list.append(lasso.get_weights()[1:])
    
#Stack into numpy array
theta_lasso = np.stack(theta_list).T

n,_ = theta_lasso.shape
plt.figure(figsize = (16,10))

for i in range(n):
    plt.plot(alpha, theta_lasso[i], 'o-', label = all_features[i],)
    
plt.xscale('log')
plt.xlabel('Log($\\alpha$)')
plt.ylabel('Coefficients')
plt.title('Lasso Paths Coefficients')
plt.legend(prop={'size': 10})
plt.axis('tight')
